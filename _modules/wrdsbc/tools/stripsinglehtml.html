[docs]def stripsinglehtml(path='index.html'):
    """
    strip markup from sphinx singlehtml files
    (rather than writing a sphinx [...]-er)
    """
    contents = None
    with open(path, 'r') as f:
        contents = f.read()

    bs = bs4.BeautifulSoup(contents)
    bs.find('body')
    bs = bs.find('body')
    [x.extract() for x in bs.find_all('a', {'class':'headerlink'})]
    bs.find('div', {'class':'sphinxsidebar'}).extract()
    bs.find('div', {'id':'indices-and-tables'}).extract()
    [x.extract() for x in bs.find_all('div', {'class':'related'})]

    return bs

[docs]def main(*args):
    import logging
    import optparse
    import sys

    prs = optparse.OptionParser(
        usage="%prog <path>",
        description="Strip markup from sphinx singlehtml renders")

    prs.add_option('-g', '--google-sites',
                   dest='google_sites',
                   action='store_true')

    prs.add_option('-v', '--verbose',
                    dest='verbose',
                    action='store_true',)
    prs.add_option('-q', '--quiet',
                    dest='quiet',
                    action='store_true',)
    prs.add_option('-t', '--test',
                    dest='run_tests',
                    action='store_true',)

    args = args and list(args) or sys.argv[1:]
    (opts, args) = prs.parse_args()

    if not opts.quiet:
        logging.basicConfig()

        if opts.verbose:
            logging.getLogger().setLevel(logging.DEBUG)

    if opts.run_tests:
        sys.argv = [sys.argv[0]] + args
        import unittest
        sys.exit(unittest.main())

    if len(args) != 1:
        raise Exception("Must specify a file path")
    path = args[0]

    import codecs
    sys.stdout = codecs.getwriter('utf-8')(sys.stdout, errors='replace')

    bs = stripsinglehtml(path)

    if opts.google_sites:
        bs = format_for_google_sites(bs)

    print(unicode(bs))



if __name__ == "__main__":
    import sys
    sys.exit(main())
